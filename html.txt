<script>
  const btn = document.querySelector('button');
  const input = document.getElementById('ltcAddress');

  btn.addEventListener('click', async () => {
    const address = input.value.trim();
    if (!address) {
      alert('Por favor ingresa tu dirección LTC');
      return;
    }

    try {
      const response = await fetch('http://localhost:3000/api/send', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ address: address, amount: 0.001 }) // ajusta cantidad aquí si quieres
      });

      const data = await response.json();

      if (data.error) {
        alert('Error: ' + data.error);
      } else {
        alert('¡Reclamo enviado con éxito! Detalles: ' + JSON.stringify(data));
      }
    } catch (err) {
      alert('Error al conectar con el servidor: ' + err.message);
    }
  });
</script>
